                    function toCamelCase(str){
                        var arr = str.replace(/_/g, '-').split('-')
                        for (let i = 1; i < arr.length; i++)
                          {
                            arr[i] = arr[i][0].toUpperCase() + arr[i].slice(1)
                          }
                        return arr.join("")
                    }
